```
function paginationUrl(p: number) {
  const url = new URLSearchParams()
  if (filterParams) {
    Object.entries(filterParams).forEach(([k, v]) => {
      if (v && k !== 'page') url.set(k, v)
    })
  }
  url.set('page', String(p))
  return `/portal/leads?${url.toString()}`
}
```
